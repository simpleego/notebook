# 가위 바위 보 게임 승부 결정
> n명이 게임을 할 경우
>
```C
// 사용하지 말 것



#include <stdio.h>

#include <stdlib.h>

#include <time.h>

 

enum Gawibawibo { GAWI = 0, BAWI, BO };

 

#define NUM_PLAYER  10

 

// 게임에서이긴사람들을저장

int WINNER_LIST[NUM_PLAYER];

int nWinners = 0;

 

// Return value

// -1 : 비김

// 아니면 승자가 낸 것을 반환

int Toss(Gawibawibo _Players[], int _Count);

 

void Report(Gawibawibo _Players[], int _Count);

 

int main()

{

       int i;

       Gawibawibo players[NUM_PLAYER];

       int winner;

 

       srand((unsigned int)time(NULL));

 

       for ( i = 0; i < NUM_PLAYER; ++i )

             players[i] = (Gawibawibo)(rand() % 3);

 

       printf("Start...\n");

       Report(players, NUM_PLAYER);

 

       winner = Toss(players, NUM_PLAYER);

 

       if ( winner == -1 )

             printf("Even game\n");

       else

             for ( i = 0; i < nWinners; ++i )

                    printf("Winner %d\n", WINNER_LIST[i]);

 

       return 0;

}

 

int Toss(Gawibawibo _Players[], int _Count)

{

       int flag_Gawi = 0;

       int flag_Bawi = 0;

       int flag_Bo = 0;

 

       int i;

 

       // 1차로무승부인지부터확인한다

       // 무승부는전부다한가지만냈을경우와

       // 가위/바위/보가모두나왔을경우이다

      

       for ( i = 0; i < _Count; ++i )

       {

             if ( _Players[i] == GAWI )

                    flag_Gawi = 1;

             else if ( _Players[i] == BAWI )

                    flag_Bawi = 1;

             else

                    flag_Bo = 1;

       }

       // 한가지만나왔을경우무승부

       if ( ( flag_Gawi + flag_Bawi + flag_Bo ) == 1 )

             return -1;

       // 세가지다나왔을경우도무승부

       if ( flag_Gawi & flag_Bawi & flag_Bo )

             return -1;

 

       // 승부결정

       nWinners = 0;

       // 가위가없는경우이므로나머지는모두바위아니면보이다

       // 따라서BO 낸사람들이승리

       if ( flag_Gawi == 0 )

       {

             for ( i = 0; i < NUM_PLAYER; ++i )

                    // 보낸사람들을승자리스트에저장

                    if ( _Players[i] == BO )

                    {

                           WINNER_LIST[nWinners++] = i;

                    }

             return 0;

       }

       else if ( flag_Bawi == 0 )

       {

             for ( i = 0; i < NUM_PLAYER; ++i )

                    if ( _Players[i] == GAWI )

                    {

                           WINNER_LIST[nWinners++] = i;

                    }

             return 1;

       }

       else

       {

             for ( i = 0; i < NUM_PLAYER; ++i )

                    if ( _Players[i] == BAWI )

                    {

                           WINNER_LIST[nWinners++] = i;

                    }

             return 2;

       }

 

       return 0;

}

 

void Report(Gawibawibo _Players[], int _Count)

{

       int i;

 

       for ( i = 0; i < NUM_PLAYER; ++i )

       {

             if ( _Players[i] == GAWI )

                    printf("Player %d : 가위\n", i);

             else if ( _Players[i] == BAWI )

                    printf("Player %d : 바위\n", i);

             else if ( _Players[i] == BO )

                    printf("Player %d : 보\n", i);

             else

                    printf("Player %d : Die\n", i);

       }

       printf("-------------------------------------\n");

}

 ```
